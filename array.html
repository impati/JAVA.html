   public class Main {

      public static void main(String[] args)  {
        //배열의 선언.
        //자료형 []배열이름 = new 자료형[크기].
      	int []Arr = new int[100];
      	//배열의 길이.
      	System.out.println(Arr.length);
      	//배열의 인덱스.. 0부터 array.lenght-1 까지의 인덱스를 가지고 이를통해 배열에 접근한다.
      	for(int i=0;i<Arr.length;i++) {
          System.out.println(Arr[i]);
    	}
      //인덱스 0이 배열의 첫번째 값임을 유의한다.
  }
